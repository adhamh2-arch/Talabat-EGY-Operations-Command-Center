"/**\n * ScoringEngine.gs — cluster + vendor risk scoring on the new consolidated data.\n * processAllClusters(feedData) consumes feedData.byCluster (from DataReader).\n * v5.1: topOffenders now also carry isKey/isTgo/isFood/chainId for the 31-col tracker + action tool.\n */\n\nfunction processAllClusters(feedData) {\n  var clusterResults = [];\n\n  Object.keys(CLUSTERS).forEach(function(ck) {\n    var agg = feedData.byCluster[ck];\n    if (agg) clusterResults.push(scoreCluster_(ck, agg));\n  });\n\n  // stable, risk-first ordering for dashboards\n  clusterResults.sort(function(a, b) { return b.riskScore - a.riskScore; });\n\n  var totalAllOrders = 0, totalFailed = 0, totalLostGmv = 0, totalPartialRefund = 0, totalFaultCases = 0, weightedRate = 0;\n  clusterResults.forEach(function(c) {\n    totalAllOrders += c.allOrders;\n    totalFailed += c.failedOrders;\n    totalLostGmv += c.lostGmv;\n    totalPartialRefund += c.partialRefund;\n    totalFaultCases += c.vendorFaultCases;\n    weightedRate += (c.vendorFailRate / 100) * c.allOrders;\n  });\n  var overallFailRate = totalAllOrders > 0 ? weightedRate / totalAllOrders : 0;\n\n  return {\n    date: getCairoDate(0),\n    mtdRange: getMtdRange(),\n    clusterResults: clusterResults,\n    warnings: feedData.warnings || [],\n    health: feedData.health || [],\n    summary: {\n      totalAllOrders: totalAllOrders,\n      totalFailed: totalFailed,\n      totalLostGmv: Math.round(totalLostGmv),\n      totalPartialRefund: Math.round(totalPartialRefund),\n      totalVendorFaultCases: totalFaultCases,\n      overallFailRate: +(overallFailRate * 100).toFixed(3),\n      criticalClusters: clusterResults.filter(function(c) { return c.riskLevel === 'critical'; }).length,\n      highRiskClusters: clusterResults.filter(function(c) { return c.riskLevel === 'high'; }).length,\n      clustersAboveTarget: clusterResults.filter(function(c) { return c.deviation > 0; }).map(function(c) { return c.name; })\n    },\n    riskRegister: clusterResults\n      .filter(function(c) { return c.riskScore >= 26; })\n      .map(function(c) {\n        return {\n          cluster: c.name, riskScore: c.riskScore, riskLevel: c.riskLevel,\n          failRate: c.vendorFailRate + '%', target: c.targetLabel,\n          deviation: c.deviation.toFixed(2) + '%', lostGmv: c.lostGmv,\n          vendorFaultCases: c.vendorFaultCases,\n          topOffender: c.topOffenders.length > 0 ? c.topOffenders[0].vendor : 'N/A'\n        };\n      })\n  };\n}\n\nfunction scoreCluster_(clusterKey, agg) {\n  var def = CLUSTERS[clusterKey];\n  var t = agg.totals;\n\n  var allOrders = t.allOrders;\n  var failedOrders = t.failedOrders;\n  var vendorFailRate = t.vendorFailRate; // fraction\n  var target = def.target;\n  var deviation = vendorFailRate - target;\n  var deviationPct = target > 0 ? (deviation / target) * 100 : 0;\n\n  // ---- cluster risk score (0-100) ----\n  var riskScore = 0;\n  if (deviation > 0) riskScore += Math.min(SCORING_WEIGHTS.deviation, deviationPct * 0.8);\n  riskScore += Math.min(SCORING_WEIGHTS.vendorFaultCases, (t.vendorFaultCases / 500) * SCORING_WEIGHTS.vendorFaultCases);\n  riskScore += Math.min(SCORING_WEIGHTS.orderVolume, (allOrders / 10000) * SCORING_WEIGHTS.orderVolume);\n  riskScore += Math.min(SCORING_WEIGHTS.financialImpact, (t.lostGmv + t.partialRefund) / 20000);\n  riskScore = Math.min(100, Math.round(riskScore));\n  var riskLevel = classifyRiskLevel_(riskScore);\n\n  // ---- per-vendor risk scoring + top offenders ----\n  var scoredVendors = agg.vendors.map(function(v) {\n    v.riskScore = scoreVendor_(v, target);\n    v.riskLevel = classifyRiskLevel_(v.riskScore);\n    return v;\n  });\n\n  var offenders = scoredVendors.filter(function(v) {\n    return v.failedOrders > 0 || v.vendorFaultCases > 0;\n  }).sort(function(a, b) {\n    if (b.riskScore !== a.riskScore) return b.riskScore - a.riskScore;\n    return b.failedOrders - a.failedOrders;\n  });\n\n  var topOffenders = offenders.slice(0, TOP_OFFENDERS_LIMIT).map(function(v) {\n    return {\n      vendorId: v.vendorId, vendor: v.vendor, city: v.city,\n      allOrders: v.allOrders, failedOrders: v.failedOrders,\n      failRate: v.failRate, lostGmv: v.lostGmv, partialRefund: v.partialRefund,\n      amName: v.amName, amEmail: v.amEmail, teamLeaderName: v.teamLeaderName,\n      teamLeaderEmail: def.tlEmail,\n      segment: v.segment,\n      isKey: v.isKey, isTgo: v.isTgo, isFood: v.isFood, chainId: v.chainId,\n      lateDeliveryCases: v.lateDeliveryCases, missingItemCases: v.missingItemCases,\n      foodQualityCases: v.qualityCases, qualityCases: v.qualityCases,\n      vendorFaultCases: v.vendorFaultCases,\n      topReason: v.topReason, rootCause: v.rootCause,\n      riskScore: v.riskScore, riskLevel: v.riskLevel\n    };\n  });\n\n  var cityDetails = agg.cities.map(function(c) {\n    return {\n      city: c.city, allOrders: c.allOrders, failedOrders: c.failedOrders,\n      vendorFailRate: c.vendorFailRate, lostGmv: c.lostGmv, partialRefund: c.partialRefund\n    };\n  });\n\n  return {\n    cluster: clusterKey, key: clusterKey, name: def.name,\n    segment: def.segment,\n    tlName: def.tlName, tlEmail: def.tlEmail,\n    target: target, targetLabel: def.targetLabel,\n    allOrders: allOrders, failedOrders: failedOrders, successfulOrders: allOrders - failedOrders,\n    lostGmv: Math.round(t.lostGmv), partialRefund: Math.round(t.partialRefund),\n    financialImpact: Math.round(t.lostGmv + t.partialRefund),\n    vendorFailRate: +(vendorFailRate * 100).toFixed(3),\n    deviation: +(deviation * 100).toFixed(3),\n    deviationPct: +deviationPct.toFixed(1),\n    vendorFaultCases: t.vendorFaultCases,\n    lateDeliveryCases: t.lateDeliveryCases, missingItemCases: t.missingItemCases,\n    qualityCases: t.qualityCases, foodQualityCases: t.qualityCases,\n    riskScore: riskScore, riskLevel: riskLevel,\n    vendorCount: agg.vendors.length,\n    offenderCount: offenders.length,\n    cities: cityDetails,\n    topOffenders: topOffenders,\n    failReasons: agg.failReasons.slice(0, 12)\n  };\n}\n\nfunction scoreVendor_(v, target) {\n  var rate = v.allOrders > 0 ? v.failedOrders / v.allOrders : 0;\n  var relDev = target > 0 ? Math.max(0, (rate - target) / target) : 0;\n  var s = 0;\n  s += Math.min(35, relDev * 18);                                  // severity over target\n  s += Math.min(30, v.failedOrders / 2);                           // absolute failed volume\n  s += Math.min(20, v.vendorFaultCases * 1.5);                     // vendor-fault case load\n  s += Math.min(15, (v.lostGmv + v.partialRefund) / 3000);         // financial impact\n  return Math.min(100, Math.round(s));\n}\n\nfunction classifyRiskLevel_(score) {\n  if (score >= 76) return 'critical';\n  if (score >= 51) return 'high';\n  if (score >= 26) return 'medium';\n  return 'low';\n}\n"
