"/**\n * TrackerWriter.gs — v5.2 (31-column live schema + Chain ID at AF).\n * Writes daily offenders into the Vendor Action Tracker (cols A–AE / 1–31) AND now\n * stamps the Chain ID into AF (col 32) at daily-write time, sourced straight from the\n * feeder data (DataReader captures chainId -> ScoringEngine.topOffenders.chainId).\n * This removes the reliance on ActionTool.refreshChainIds() as the only way AF gets filled,\n * so the Tuesday portal action agent always has a real Chain ID on freshly-written rows.\n * The remaining action columns AG–AI (33–35, Requested Action / Last Seen Status /\n * Action Taken) are OWNED by ActionTool.gs and are never touched here.\n */\nvar TRACKER_COLS = 31; i < existing.length; i++) {\n    if (String(existing[i][0]) === today) {\n      return { rowsAdded: 0, message: 'Skipped — already tracked for ' + today };\n    }\n  }\n\n  var history = {};\n  for (var i = 1; i < existing.length; i++) {\n    var vn = String(existing[i][2]);\n    if (vn) history[vn] = (history[vn] || 0) + 1;\n  }\n\n  var rows = [];\n  scoredOutput.clusterResults.forEach(function(cluster) {\n    (cluster.topOffenders || []).forEach(function(v) {\n      var appearances = (history[v.vendor] || 0) + 1;\n      rows.push([\n        today,\n        v.vendorId || '',\n        v.vendor,\n        v.city,\n        cluster.name,\n        v.amName || '',\n        v.amEmail || '',\n        v.teamLeaderName || '',\n        v.isKey || '',\n        v.isTgo || '',\n        v.isFood || '',\n        v.allOrders || 0,\n        v.failedOrders,\n        v.lostGmv,\n        v.failRate + '%',\n        v.foodQualityCases || 0,\n        v.lateDeliveryCases || 0,\n        v.missingItemCases || 0,\n        v.vendorFaultCases || 0,\n        (v.riskScore != null ? v.riskScore : cluster.riskScore),\n        String(v.riskLevel || cluster.riskLevel).toUpperCase(),\n        appearances,\n        getRepeatFlag_(appearances),\n        '', '', '', '', '', '', '',\n        'Open',\n        v.chainId || ''   // AF (col 32) — Chain ID from feeder, for the Tuesday portal action agent\n      ]);\n    });\n  });\n\n  if (rows.length === 0) {\n    return { rowsAdded: 0, message: 'No offenders to track' };\n  }\n\n  rows.sort(function(a, b) {\n    if (b[21] !== a[21]) return b[21] - a[21];   // appearances desc\n    return b[18] - a[18];                         // vendor fault cases desc\n  });\n\n  // make sure the Chain ID column (AF/32) exists before we write into it\n  if (sheet.getMaxColumns() < TRACKER_WRITE_COLS) {\n    sheet.insertColumnsAfter(sheet.getMaxColumns(), TRACKER_WRITE_COLS - sheet.getMaxColumns());\n  }\n\n  var startRow = sheet.getLastRow() + 1;\n  // write A–AF (32 cols): the 31-col core block + Chain ID. AG–AI stay owned by ActionTool.\n  sheet.getRange(startRow, 1, rows.length, TRACKER_WRITE_COLS).setValues(rows);\n\n  formatTrackerRows_(sheet, startRow, rows.length);\n  applyTrackerProtection_(sheet);\n\n  return { rowsAdded: rows.length, message: rows.length + ' vendors tracked for ' + today };\n}\n\nfunction initTrackerSheet_(sheet) {\n  var headers = [\n    'Run Date', 'Vendor ID', 'Vendor Name', 'City', 'Cluster',\n    'AM Name', 'AM Email', 'Team Leader', 'Is Key VIP', 'Is TGO', 'Is Food',\n    'Net Orders', 'Net Failed Orders', 'Lost GMV (EGP)', 'Net Fail Rate %',\n    'Order Quality Cases', 'Late Delivery Cases', 'Missing Item Cases', 'Vendor Fault Cases',\n    'Risk Score', 'Risk Level', 'Appearances MTD', 'Repeat Flag',\n    'Alert Sent', 'Alert Date', 'Escalation Sent', 'Escalation Date',\n    'Performance Improved', 'Action Plan Shared', 'Action Plan Details', 'Status'\n  ];\n\n  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);\n\n  var hdr = sheet.getRange(1, 1, 1, headers.length);\n  hdr.setBackground(THEME.primary)\n     .setFontColor(THEME.headerText)\n     .setFontWeight('bold')\n     .setFontSize(10)\n     .setHorizontalAlignment('center')\n     .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP);\n\n  // segment flags + case-data + workflow header tints\n  sheet.getRange(1, 9, 1, 3).setBackground('#8A5A00');\n  sheet.getRange(1, 16, 1, 4).setBackground('#E65100');\n  sheet.getRange(1, 24, 1, 8).setBackground('#E65100');\n\n  var widths = [100, 90, 240, 120, 90, 140, 200, 140, 80, 70, 70, 90, 90, 110, 80, 90, 90, 90, 90, 70, 80, 90, 80, 80, 90, 90, 100, 100, 110, 260, 90];\n  for (var i = 0; i < widths.length; i++) sheet.setColumnWidth(i + 1, widths[i]);\n\n  sheet.setFrozenRows(1);\n  sheet.setFrozenColumns(3);\n\n  var yn = SpreadsheetApp.newDataValidation().requireValueInList(['', 'Yes', 'No']).setAllowInvalid(false).build();\n  sheet.getRange('X2:X1000').setDataValidation(yn);   // Alert Sent\n  sheet.getRange('Z2:Z1000').setDataValidation(yn);   // Escalation Sent\n\n  var improvedRule = SpreadsheetApp.newDataValidation().requireValueInList(['', 'Yes', 'No', 'N/A']).setAllowInvalid(false).build();\n  sheet.getRange('AB2:AB1000').setDataValidation(improvedRule);  // Performance Improved\n\n  var actionRule = SpreadsheetApp.newDataValidation().requireValueInList(['', 'Yes', 'No', 'Partial']).setAllowInvalid(false).build();\n  sheet.getRange('AC2:AC1000').setDataValidation(actionRule);    // Action Plan Shared\n\n  var statusRule = SpreadsheetApp.newDataValidation().requireValueInList(['Open', 'In Progress', 'Resolved', 'Escalated', 'No Response']).setAllowInvalid(false).build();\n  sheet.getRange('AE2:AE1000').setDataValidation(statusRule);    // Status\n}\n\nfunction getRepeatFlag_(count) {\n  if (count >= 6) return 'CRITICAL';\n  if (count >= 4) return 'HIGH';\n  if (count >= 2) return 'MEDIUM';\n  return 'NEW';\n}\n\nfunction formatTrackerRows_(sheet, startRow, numRows) {\n  var range = sheet.getRange(startRow, 1, numRows, TRACKER_COLS);\n  range.setFontSize(9).setBorder(true, true, true, true, true, true, '#E0E0E0', SpreadsheetApp.BorderStyle.SOLID);\n\n  sheet.getRange(startRow, 12, numRows, 1).setNumberFormat('#,##0');    // Net Orders\n  sheet.getRange(startRow, 13, numRows, 1).setNumberFormat('#,##0');    // Net Failed Orders\n  sheet.getRange(startRow, 14, numRows, 1).setNumberFormat('#,##0.00'); // Lost GMV\n  sheet.getRange(startRow, 16, numRows, 4).setNumberFormat('#,##0');    // case columns\n\n  sheet.getRange(startRow, 24, numRows, 8).setBackground('#E3F2FD');    // workflow block\n\n  for (var i = 0; i < numRows; i++) {\n    var row = startRow + i;\n    var vals = sheet.getRange(row, 1, 1, TRACKER_COLS).getValues()[0];\n\n    var flagCell = sheet.getRange(row, 23);  // Risk Level\n    switch (String(vals[20]).toUpperCase()) {\n      case 'CRITICAL': riskCell.setBackground(THEME.critical).setFontColor('#FFF'); break;\n      case 'HIGH':     riskCell.setBackground(THEME.high).setFontColor('#FFF'); break;\n      case 'MEDIUM':   riskCell.setBackground(THEME.medium).setFontColor('#000'); break;\n      case 'LOW':      riskCell.setBackground(THEME.low).setFontColor('#FFF'); break;\n    }\n\n    var statusCell = sheet.getRange(row, 31);  // Status\n    switch (vals[30]) {\n      case 'Open':        statusCell.setBackground('#FFCDD2').setFontColor('#B71C1C'); break;\n      case 'In Progress': statusCell.setBackground('#FFF9C4').setFontColor('#F57F17'); break;\n      case 'Resolved':    statusCell.setBackground('#C8E6C9').setFontColor('#1B5E20'); break;\n      case 'Escalated':   statusCell.setBackground('#CE93D8').setFontColor('#4A148C'); break;\n      case 'No Response': statusCell.setBackground('#FFCCBC').setFontColor('#BF360C'); break;\n    }\n  }\n}\n\nfunction applyTrackerProtection_(sheet) {\n  var protections = sheet.getProtections(SpreadsheetApp.ProtectionType.SHEET);\n  protections.forEach(function(p) { p.remove(); });\n\n  var protection = sheet.protect()\n    .setDescription('Historical data locked — workflow + action columns (X–AI) are editable')\n    .setWarningOnly(true);\n\n  var lastRow = Math.max(sheet.getLastRow(), 2);\n  protection.setUnprotectedRanges([sheet.getRange('X2:AI' + lastRow)]);\n}\n\nfunction getWeeklyOffenders_(sheet) {\n  var today = new Date();\n  var dayOfWeek = parseInt(Utilities.formatDate(today, 'Africa/Cairo', 'u'));\n  var daysBack = (dayOfWeek >= 2) ? (dayOfWeek - 2) : (dayOfWeek + 5);\n  var tuesdayDate = new Date(today);\n  tuesdayDate.setDate(today.getDate() - daysBack);\n  var tuesdayStr = Utilities.formatDate(tuesdayDate, 'Africa/Cairo', 'yyyy-MM-dd');\n\n  var data = sheet.getDataRange().getValues();\n  var offenders = [];\n  for (var i = 1; i < data.length; i++) {\n    var runDate = String(data[i][0]);\n    if (runDate >= tuesdayStr) {\n      offenders.push({\n        row: i + 1,\n        runDate: runDate,\n        vendorId: String(data[i][1]),\n        vendor: String(data[i][2]),\n        city: String(data[i][3]),\n        cluster: String(data[i][4]),\n        amName: String(data[i][5]),\n        amEmail: String(data[i][6]),\n        teamLeaderName: String(data[i][7]),\n        allOrders: Number(data[i][11]) || 0,\n        failedOrders: Number(data[i][12]) || 0,\n        lostGmv: Number(data[i][13]) || 0,\n        failRate: String(data[i][14]),\n        foodQualityCases: Number(data[i][15]) || 0,\n        lateDeliveryCases: Number(data[i][16]) || 0,\n        missingItemCases: Number(data[i][17]) || 0,\n        vendorFaultCases: Number(data[i][18]) || 0,\n        riskScore: Number(data[i][19]) || 0,\n        riskLevel: String(data[i][20]),\n        alertSent: String(data[i][23]),\n        escalationSent: String(data[i][25]),\n        performanceImproved: String(data[i][27]),\n        actionPlanShared: String(data[i][28]),\n        status: String(data[i][30]),\n        chainId: String(data[i][31] || '')\n      });\n    }\n  }\n  return offenders;\n}\n\nfunction markTuesdayAlerts_(sheet, vendorRows) {\n  var today = getCairoDate(0);\n  vendorRows.forEach(function(v) {\n    sheet.getRange(v.row, 24).setValue('Yes');
