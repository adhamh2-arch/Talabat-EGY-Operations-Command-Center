"/**\n * ActionTool.gs — talabat EGY Ops Command Center: Vendor Status Action module (v5.2, July 2026)\n *\n * Purpose: turn the Vendor Action Tracker into the monitor + action dashboard for the\n * weekly (Tuesday) portal status-update workflow driven by the Chrome action agent.\n *\n * v5.2 REVAMPED ACTION BLOCK (AF–AI):\n *   AF Chain ID\n *   AG Requested Action (auto)      — auto-computed from VFR% + Action Plan Shared (see AG formula)\n *   AH Last Seen Status (found)     — current portal status the agent found at check time\n *   AI Action Taken (set)           — what the agent changed it to (or \"skipped — permanent Busy\")\n *\n * Approved vocab (Project Structure tab, col M = Yes): Open (restore, no reason) and\n * Hidden + TOO_MANY_REJECTED_ORDERS (corrective). Nothing else is permitted.\n *\n * NOTE: standalone script (not container-bound in the classic sense) — exposes callable\n * functions run from the editor / menu / triggers rather than doing the portal work itself\n * (the Chrome action agent in Cowork drives the portal).\n */\n\n// ---- Tracker column map (1-based). Live 31-col schema + 4 action columns. ----\nvar TCOL = {\n  runDate: 1, vendorId: 2, vendorName: 3, city: 4, cluster: 5,\n  amName: 6, amEmail: 7, teamLeader: 8, isKeyVip: 9, isTgo: 10, isFood: 11,\n  netOrders: 12, netFailedOrders: 13, lostGmv: 14, netFailRate: 15,\n  orderQualityCases: 16, lateDeliveryCases: 17, missingItemCases: 18, vendorFaultCases: 19,\n  riskScore: 20, riskLevel: 21, appearancesMtd: 22, repeatFlag: 23,\n  alertSent: 24, alertDate: 25, escalationSent: 26, escalationDate: 27,\n  performanceImproved: 28, actionPlanShared: 29, actionPlanDetails: 30, status: 31,\n  // --- revamped action columns (v5.2) ---\n  chainId: 32, requestedAction: 33, lastSeenStatus: 34, actionTaken: 35\n};\nvar TRACKER_TOTAL_COLS = 35;\n\n// Need-action rule (Adham, July 2026): offender = VFR (Net Fail Rate %) between 40% and 100%.\nvar VFR_ACTION_THRESHOLD = 40; // percent (lower bound). Upper bound 100. Tune here.\n\nvar PORTAL_STATUS_BASE = 'https://portal.talabat.com/eg/p/restaurant-management#/restaurants/';\n\n// APPROVED action combos (Project Structure col M = Yes). ONLY these two are permitted:\n//   Open  = restore to active (no reason)\n//   Hidden | TOO_MANY_REJECTED_ORDERS = the corrective action for offenders\nvar APPROVED_ACTION = 'Hidden | TOO_MANY_REJECTED_ORDERS';\nvar RESTORE_ACTION = 'Open';\n\n/**\n * AG \"Requested Action\" is AUTOMATED via this single spilling ARRAYFORMULA in AG2:\n *   - blank when there is no VFR value\n *   - \"Open\" when an Action Plan was Shared (Yes/Partial)  -> action deferred to next cycle\n *   - \"Hidden | TOO_MANY_REJECTED_ORDERS\" when VFR >= 40%\n *   - blank otherwise (VFR < 40%)\n * O = Net Fail Rate % (col 15), AC = Action Plan Shared (col 29).\n */\nfunction requestedActionFormula_() {\n  return '=ARRAYFORMULA(IF(LEN($O$2:$O)=0,\"\",' +\n         'IF(($AC$2:$AC=\"Yes\")+($AC$2:$AC=\"Partial\")>0,\"' + RESTORE_ACTION + '\",' +\n         'IF(IFERROR(VALUE(SUBSTITUTE($O$2:$O,\"%\",\"\")),0)>=' + VFR_ACTION_THRESHOLD + ',\"' + APPROVED_ACTION + '\",\"\"))))';\n}\n\n/** Build the portal branch-status URL. restaurants/=Chain ID, branches/=Vendor ID. */\nfunction buildPortalStatusUrl_(chainId, vendorId) {\n  var chain = String(chainId || '').trim();\n  var vend = String(vendorId || '').trim();\n  if (!chain) chain = vend; // fallback when Chain ID unknown\n  return PORTAL_STATUS_BASE + chain + '/branches/' + vend + '/status';\n}\n\nfunction parsePct_(s) {\n  var n = Number(String(s == null ? '' : s).replace('%', '').trim());\n  return isNaN(n) ? 0 : n;\n}\n\n/** STEP 1 — create/repair the 4 action columns (AF–AI), headers, formatting, and the AG\n *  auto-formula. Safe to re-run (idempotent). Re-run after any tracker schema migration. */\nfunction setupActionColumns() {\n  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n  var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n  if (!sheet) throw new Error('Vendor Action Tracker sheet not found');\n\n  if (sheet.getMaxColumns() < TRACKER_TOTAL_COLS) {\n    sheet.insertColumnsAfter(sheet.getMaxColumns(), TRACKER_TOTAL_COLS - sheet.getMaxColumns());\n  }\n\n  var headers = [['Chain ID',\n                  'Requested Action (auto)',\n                  'Last Seen Status (found)',\n                  'Action Taken (set)']];\n  sheet.getRange(1, TCOL.chainId, 1, 4).setValues(headers);\n\n  var hdr = sheet.getRange(1, TCOL.chainId, 1, 4);\n  hdr.setBackground('#411517')     // burgundy — distinguishes the action block\n     .setFontColor('#FFFFFF')\n     .setFontWeight('bold')\n     .setFontSize(10)\n     .setHorizontalAlignment('center')\n     .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP);\n\n  sheet.setColumnWidth(TCOL.chainId, 90);\n  sheet.setColumnWidth(TCOL.requestedAction, 230);\n  sheet.setColumnWidth(TCOL.lastSeenStatus, 240);\n  sheet.setColumnWidth(TCOL.actionTaken, 240);\n\n  // Install the AG auto-formula (clear any old manual values first so the array can spill).\n  var lastRow = Math.max(sheet.getLastRow(), 2);\n  sheet.getRange(2, TCOL.requestedAction, lastRow - 1, 1).clearContent();\n  sheet.getRange(2, TCOL.requestedAction).setFormula(requestedActionFormula_());\n\n  return 'Action columns ready: AF Chain ID · AG Requested Action (auto formula) · AH Last Seen Status (found) · AI Action Taken (set).';\n}\n\n/** Build a { vendorId -> chainId } map by scanning all 7 feeders. */\nfunction buildChainIdMap_() {\n  var map = {};\n  Object.keys(FEEDERS).forEach(function(fk) {\n    for (var t = 0; t < FEED_TAB_PREFERENCE.length; t++) {\n      var raw = readFeederTab_(fk, FEED_TAB_PREFERENCE[t]);\n      if (!raw.exists || !raw.rows.length || !raw.colMap) continue;\n      var cm = raw.colMap;\n      if (cm.chainId >= 0) {\n        raw.rows.forEach(function(r) {\n          var vid = String(cell_(r, cm.vendorId) || '').trim();\n          var cid = String(cell_(r, cm.chainId) || '').trim();\n          if (vid && cid && !map[vid]) map[vid] = cid;\n        });\n      }\n      break; // first tab that has rows wins\n    }\n  });\n  return map;\n}\n\n/** STEP 2 — fill the Chain ID column for any row that has a Vendor ID but no Chain ID. */\nfunction refreshChainIds() {\n  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n  var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n  var last = sheet.getLastRow();\n  if (last < 2) return 'No data rows.';\n  __FEEDER_TAB_CACHE = {};\n  var map = buildChainIdMap_();\n  var ids = sheet.getRange(2, TCOL.vendorId, last - 1, 1).getValues();\n  var chains = sheet.getRange(2, TCOL.chainId, last - 1, 1).getValues();\n  var filled = 0, missing = 0;\n  for (var i = 0; i < ids.length; i++) {\n    var vid = String(ids[i][0] || '').trim();\n    if (!vid) continue;\n    if (String(chains[i][0] || '').trim()) continue;\n    if (map[vid]) { chains[i][0] = map[vid]; filled++; }\n    else missing++;\n  }\n  sheet.getRange(2, TCOL.chainId, last - 1, 1).setValues(chains);\n  return 'Chain IDs filled: ' + filled + ' | still missing (not found in feeders): ' + missing;\n}\n\n/**\n * STEP 3 — return the need-action queue for the Chrome agent.\n * Offender = VFR between 40% and 100%. `executable` = the auto Requested Action is the\n * corrective one (Hidden | TOO_MANY_REJECTED_ORDERS). Rows whose Requested Action is \"Open\"\n * (an Action Plan was shared) are returned as deferred (executable=false).\n */\nfunction getNeedActionQueue() {\n  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n  var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n  var last = sheet.getLastRow();\n  var out = [];\n  if (last < 2) return out;\n  var data = sheet.getRange(2, 1, last - 1, TRACKER_TOTAL_COLS).getValues();\n  for (var i = 0; i < data.length; i++) {\n    var r = data[i];\n    var vfr = parsePct_(r[TCOL.netFailRate - 1]);\n    if (vfr < VFR_ACTION_THRESHOLD || vfr > 100) continue;\n    var requested = String(r[TCOL.requestedAction - 1] || '').trim();\n    out.push({\n      row: i + 2,\n      vendorId: String(r[TCOL.vendorId - 1] || '').trim(),\n      chainId: String(r[TCOL.chainId - 1] || '').trim(),\n      vendor: String(r[TCOL.vendorName - 1] || ''),\n      city: String(r[TCOL.city - 1] || ''),\n      riskLevel: String(r[TCOL.riskLevel - 1] || '').toUpperCase(),\n      vfr: vfr,\n      actionPlanShared: String(r[TCOL.actionPlanShared - 1] || ''),\n      requestedAction: requested,\n      executable: requested.indexOf('Hidden') === 0,   // corrective; \"Open\" = deferred\n      deferred: requested === RESTORE_ACTION,\n      lastSeenStatus: String(r[TCOL.lastSeenStatus - 1] || ''),\n      actionTaken: String(r[TCOL.actionTaken - 1] || ''),\n      url: buildPortalStatusUrl_(r[TCOL.chainId - 1], r[TCOL.vendorId - 1])\n    });\n  }\n  return out;\n}\n\n/** Convenience: log the queue to Stackdriver (for manual runs / debugging). */\nfunction logNeedActionQueue() {\n  var q = getNeedActionQueue();\n  Logger.log('40–100% VFR offenders: ' + q.length);\n  q.forEach(function(v) {\n    Logger.log([v.row, v.vendorId, v.chainId, v.riskLevel, v.vfr + '%',\n      v.executable ? 'ACT:' + v.requestedAction : (v.deferred ? 'DEFER (action plan shared)' : 'no action'),\n      v.url].join(' | '));\n  });\n  return q.length + ' offenders (see logs).';\n}\n\n/**\n * TAKE ACTION button — shows the current 40–100% VFR queue and portal links so you (or the\n * Chrome agent) can act. Requested Action (AG) is auto-computed by formula, so this no longer\n * stamps anything — it just surfaces the queue. Executable = \"Hidden | TOO_MANY_REJECTED_ORDERS\";\n * deferred = \"Open\" (an Action Plan was shared -> take action next cycle).\n */\nfunction prepareTakeAction() {\n  var ui = SpreadsheetApp.getUi();\n  var q = getNeedActionQueue();\n  if (q.length === 0) { ui.alert('Take Action', 'No vendors in the 40–100% VFR queue right now.', ui.ButtonSet.OK); return; }\n\n  var act = q.filter(function(v) { return v.executable; });\n  var defer = q.filter(function(v) { return v.deferred; });\n  var links = act.map(function(v) { return '• ' + v.vendor + '  (' + v.vfr + '%)  ->  ' + v.url; });\n\n  Logger.log('prepareTakeAction — to act: ' + act.length + ', deferred (action plan): ' + defer.length + '\\n' + links.join('\\n'));\n  ui.alert('Take Action — ' + q.length + ' offender(s) in queue',\n    'To ACT this cycle (Hidden | TOO_MANY_REJECTED_ORDERS): ' + act.length + '\\n' +\n    'DEFERRED — Action Plan shared, keep Open, act next cycle: ' + defer.length + '\\n\\n' +\n    'Run the Chrome action agent to apply on the portal — use the Cowork \"Take Action\" page or the ' +\n    'Tuesday run. Portal links for the actionable vendors are in the execution log (View > Logs).\\n\\n' +\n    'Approved actions only: ' + APPROVED_ACTION + ' , or ' + RESTORE_ACTION + ' to restore.',\n    ui.ButtonSet.OK);\n  return q.length + ' offenders (' + act.length + ' to act, ' + defer.length + ' deferred).';\n}\n\n/**\n * STEP 4 — logging helper the agent calls after each portal check/update.\n *   foundStatus  -> AH Last Seen Status (found)   e.g. 'Open' / 'Busy (permanent)' / 'Hidden (...)'\n *   setStatus    -> AI Action Taken (set)         e.g. 'Set Hidden (TOO_MANY_REJECTED_ORDERS)' /\n *                                                  'Skipped — permanent Busy' / 'No change (deferred)'\n * Writes against the most recent tracker row for that Vendor ID.\n */\nfunction logPortalAction(vendorId, foundStatus, setStatus) {\n  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n  var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n  var last = sheet.getLastRow();\n  var ids = sheet.getRange(2, TCOL.vendorId, last - 1, 1).getValues();\n  var stamp = getCairoDate(0) + ' ' + Utilities.formatDate(new Date(), 'Africa/Cairo', 'HH:mm');\n  for (var i = ids.length - 1; i >= 0; i--) { // last matching row (most recent run)\n    if (String(ids[i][0]).trim() === String(vendorId).trim()) {\n      var row = i + 2;\n      if (foundStatus != null) sheet.getRange(row, TCOL.lastSeenStatus).setValue(foundStatus + ' @ ' + stamp);\n      if (setStatus != null) sheet.getRange(row, TCOL.actionTaken).setValue(setStatus + ' @ ' + stamp);\n      return 'Logged row ' + row + ': found=' + foundStatus + ' set=' + setStatus;\n    }\n  }\n  return 'Vendor ID ' + vendorId + ' not found.';\n}\n\n/** One-shot setup: create/repair the action columns (+ AG formula) and back-fill Chain IDs. */\nfunction setupActionModule() {\n  var a = setupActionColumns();\n  var b = refreshChainIds();\n  return a + '\\n' + b;\n}\n"
