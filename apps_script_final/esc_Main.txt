"/**\n * Main.gs — orchestration for the automated weekly workflow.\n *\n *  DAILY   (dailyRun)        read feeders -> ops/root-cause/risk analysis ->\n *                            update tracker with offenders -> critical email to AMs\n *  TUESDAY (tuesdayReport)   leader performance overview + vendors up for Monday action\n *  SUNDAY  (sundayReport)    MTD wrap + operations analysis to leaders; review improvements\n *  MONDAY  (mondayEscalation)check responses, escalate unresolved vendors, update tracker\n */\n\n// ============================ DAILY ============================\nfunction dailyRun() {\n  var startTime = new Date();\n  var runLog = { runDate: getCairoDate(0), status: 'RUNNING', steps: [], errors: [], emailsSent: 0, clustersProcessed: 0, totalOrders: 0, totalFailed: 0, riskSummary: '' };\n  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n\n  try {\n    Logger.log('=== Daily Run: ' + startTime.toISOString() + ' ===');\n\n    var feedData = readFeedData();\n    runLog.steps.push('Feeders Read');\n    (feedData.warnings || []).forEach(function(w) { Logger.log(w); });\n\n    var totalRows = 0;\n    (feedData.health || []).forEach(function(h) { totalRows += h.vendorRows; });\n    if (totalRows === 0) throw new Error('All feeders returned 0 vendor rows — check feeder access / IMPORTRANGE / extract refresh.');\n\n    var scoredOutput = processAllClusters(feedData);\n    runLog.steps.push('Scored (v5.0)');\n    runLog.clustersProcessed = scoredOutput.clusterResults.length;\n    runLog.totalOrders = scoredOutput.summary.totalAllOrders;\n    runLog.totalFailed = scoredOutput.summary.totalFailed;\n\n    var sheetSteps = updateAllSheets(scoredOutput, feedData);\n    runLog.steps.push('Dashboards: ' + sheetSteps.join(', '));\n\n    var trackerResult = updateVendorTracker(scoredOutput, feedData, false);\n    runLog.steps.push('Tracker: ' + trackerResult.message);\n\n    // v5.1: keep the action-tool Chain ID column populated for the Tuesday portal agent\n    try { if (typeof refreshChainIds === 'function') { refreshChainIds(); runLog.steps.push('Chain IDs refreshed'); } }\n    catch (ce) { runLog.errors.push('refreshChainIds: ' + ce.message); }\n\n    var mail = sendDailyCriticalAlerts(scoredOutput);\n    runLog.emailsSent = mail.sent;\n    runLog.steps.push('Daily critical emails: ' + mail.sent + ' sent, ' + mail.skipped + ' skipped');\n    if (mail.errors.length) runLog.errors = runLog.errors.concat(mail.errors);\n\n    runLog.riskSummary = scoredOutput.clusterResults.map(function(c) { return c.name + ':' + c.riskLevel + '(' + c.riskScore + ')'; }).join(', ');\n    if (feedData.warnings && feedData.warnings.length) runLog.errors = runLog.errors.concat(feedData.warnings);\n    runLog.status = 'SUCCESS';\n\n    // proactively flag empty/stale feeders to the ops owner\n    var empty = (feedData.health || []).filter(function(h) { return h.vendorRows === 0; });\n    if (empty.length) {\n      sendAlertEmail('Command Center — ' + empty.length + ' feeder(s) need a refresh (' + getCairoDate(0) + ')',\n        'These clusters returned no rows today:\\n\\n' + empty.map(function(h) { return '• ' + h.cluster + '  (' + h.feeder + ')'; }).join('\\n') +\n        '\\n\\nPlease refresh the feeder extract(s). Full health:\\n' + (feedData.health || []).map(function(h) { return h.cluster + ': ' + h.vendorRows + ' vendors via ' + h.source; }).join('\\n'));\n    }\n\n  } catch (e) {\n    runLog.status = 'FAILED';\n    runLog.errors.push(e.message);\n    Logger.log('FATAL: ' + e.message + '\\n' + e.stack);\n    sendAlertEmail('Command Center daily run FAILED — ' + getCairoDate(0), 'Error: ' + e.message + '\\n\\nSteps: ' + runLog.steps.join(', '));\n  }\n\n  finishRun_(ss, runLog, startTime);\n  return runLog;\n}\n\n// ============================ TUESDAY ============================\nfunction tuesdayReport() {\n  Logger.log('=== Tuesday Leader Report ===');\n  try {\n    var feedData = readFeedData();\n    var scoredOutput = processAllClusters(feedData);\n\n    var result = sendTuesdayLeaderReports(scoredOutput);\n    Logger.log('Tuesday reports: ' + result.sent + '/' + result.total);\n\n    // mark today's shortlisted offenders as alerted in the tracker\n    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n    var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n    if (sheet) {\n      var offenders = getWeeklyOffenders_(sheet);\n      var today = offenders.filter(function(v) { return v.runDate === getCairoDate(0); });\n      if (today.length) markTuesdayAlerts_(sheet, today);\n    }\n    if (result.errors.length) Logger.log('Errors: ' + result.errors.join('; '));\n  } catch (e) {\n    Logger.log('Tuesday report FAILED: ' + e.message);\n    sendAlertEmail('Command Center Tuesday report FAILED — ' + getCairoDate(0), 'Error: ' + e.message);\n  }\n}\n\n// ============================ SUNDAY ============================\nfunction sundayReport() {\n  Logger.log('=== Sunday Weekly Report ===');\n  try {\n    var feedData = readFeedData();\n    var scoredOutput = processAllClusters(feedData);\n\n    var result = sendSundayWeeklyReports(scoredOutput);\n    Logger.log('Sunday wraps: ' + result.sent + '/' + result.total);\n\n    // review improvements vs earliest tracked snapshot this week\n    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n    var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n    if (sheet) {\n      var offenders = getWeeklyOffenders_(sheet);\n      var alerted = offenders.filter(function(v) { return v.alertSent === 'Yes'; });\n      if (alerted.length) {\n        var current = {};\n        scoredOutput.clusterResults.forEach(function(c) { (c.topOffenders || []).forEach(function(v) { current[v.vendor] = v; }); });\n        var byName = {};\n        alerted.forEach(function(v) { (byName[v.vendor] = byName[v.vendor] || []).push(v); });\n        var improved = {};\n        Object.keys(byName).forEach(function(name) {\n          var earliest = byName[name][0];\n          var now = current[name];\n          if (!now) { improved[name] = true; return; }\n          improved[name] = (now.failedOrders < earliest.failedOrders) || (now.vendorFaultCases < earliest.vendorFaultCases);\n        });\n        markSundayReview_(sheet, alerted, improved);\n      }\n    }\n    if (result.errors.length) Logger.log('Errors: ' + result.errors.join('; '));\n  } catch (e) {\n    Logger.log('Sunday report FAILED: ' + e.message);\n    sendAlertEmail('Command Center Sunday report FAILED — ' + getCairoDate(0), 'Error: ' + e.message);\n  }\n}\n\n// ============================ MONDAY ============================\nfunction mondayEscalation() {\n  Logger.log('=== Monday Escalation ===');\n  try {\n    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);\n    var sheet = ss.getSheetByName(SHEET_NAMES.vendorTracker);\n    if (!sheet) { Logger.log('No tracker sheet.'); return; }\n\n    var offenders = getWeeklyOffenders_(sheet);\n    var escalationList = offenders.filter(function(v) {\n      return v.alertSent === 'Yes' && v.performanceImproved !== 'Yes' && v.actionPlanShared !== 'Yes' && v.actionPlanShared !== 'Partial';\n    });\n    if (escalationList.length === 0) { Logger.log('Nothing to escalate — all improved or have plans.'); return; }\n\n    var result = sendMondayEscalations(escalationList, getMtdRange());\n    Logger.log('Escalations: ' + result.sent + ' sent, ' + result.skipped + ' skipped');\n    if (result.sent > 0) markMondayEscalations_(sheet, escalationList);\n    if (result.errors.length) Logger.log('Errors: ' + result.errors.join('; '));\n  } catch (e) {\n    Logger.log('Monday escalation FAILED: ' + e.message);\n    sendAlertEmail('Command Center Monday escalation FAILED — ' + getCairoDate(0), 'Error: ' + e.message);\n  }\n}\n\n// ============================ shared ============================\nfunction finishRun_(ss, runLog, startTime) {\n  runLog.durationSeconds = Math.round((new Date() - startTime) / 1000);\n  try {\n    appendMonitoringLog_(ss, runLog);\n    appendAuditTrail_(ss, runLog);\n  } catch (logErr) {\n    Logger.log('Log write failed: ' + logErr.message);\n  }\n  Logger.log('=== Done: ' + runLog.status + ' in ' + runLog.durationSeconds + 's ===');\n}\n\n// ============================ test harness ============================\n// Runs the daily + weekly routines but redirects EVERY email to ALERT_EMAIL\n// (adham.h.2@talabat.com) with a [TEST -> original] subject. Sunday weekly stays in Cowork (Looker).');\n}\n\nfunction removeAllTriggers() {\n  var t = ScriptApp.getProjectTriggers();\n  t.forEach(function(x) { ScriptApp.deleteTrigger(x); });\n  Logger.log('Removed ' + t.length + ' trigger(s)');\n}\n\n// ============================ menu ============================\nfunction onOpen() {\n  SpreadsheetApp.getUi().createMenu('🟠 Command Center')\n    .addItem('▶ Run Daily Pipeline Now', 'dailyRun')\n    .addSeparator()\n    .addItem('📧 Send Tuesday Leader Report', 'manualTuesdayReport')\n    .addItem('📝 Send Sunday Weekly Report', 'manualSundayReport')\n    .addItem('🚨 Send Monday Escalation', 'manualMondayEscalation')\n    .addSeparator()\n    .addItem('🩺 Check Feeder Health', 'checkFeederHealth')\n    .addSeparator()\n    .addItem('🎯 Take Action — prep offenders (40–100% VFR)', 'prepareTakeAction')\n    .addSeparator()\n    .addItem('🧪 Email mode — show current', 'showEmailMode')\n    .addItem('      ↳ Switch to TEST (emails to me only)', 'enableTestMode')\n    .addItem('      ↳ GO LIVE (emails to real recipients)', 'disableTestMode')\n    .addSeparator()\n    .addItem('⏰ Enable All Triggers', 'setupWeeklyTriggers')\n    .addItem('⏹ Disable All Triggers', 'removeAllTriggers')\n    .addToUi();\n}\n\nfunction manualTuesdayReport() { confirmAndRun_('Send Tuesday Leader Report?', 'Performance overview + action shortlist will be emailed to Team Leaders.', 'tuesdayReport'); }\n\nfunction confirmAndRun_(title, msg, fnName) {\n  var ui = SpreadsheetApp.getUi();\n  var r = ui.alert(title, msg + '\\n\\nFrom: ' + SEND_AS_EMAIL + '  ·  CC: ' + CC_EMAIL + '\\n\\nProceed?', ui.ButtonSet.YES_NO);\n  if (r !== ui.Button.YES) { ui.alert('Cancelled', 'Nothing was sent.', ui.ButtonSet.OK);
